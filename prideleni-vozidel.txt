<script type="module">
import { initializeApp } from "https://www.gstatic.com/firebasejs/12.13.0/firebase-app.js";
import { getFirestore, collection, getDocs, doc, setDoc, getDoc } from "https://www.gstatic.com/firebasejs/12.13.0/firebase-firestore.js";
import { getAuth, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/12.13.0/firebase-auth.js";

const firebaseConfig = {
    apiKey: "AIzaSyAwRV2OuIhA-oyg295gIv5-4zfzO4qb9MY",
    authDomain: "alces-d505c.firebaseapp.com",
    projectId: "alces-d505c",
    storageBucket: "alces-d505c.firebasestorage.app",
    messagingSenderId: "318520073072",
    appId: "1:318520073072:web:53ab5c0033f60e96bbcdb5"
};

const app = initializeApp(firebaseConfig);
const db = getFirestore(app);
const auth = getAuth(app);

let currentUserEmail = "";
let currentUserName = "";

onAuthStateChanged(auth, async (user) => {
    if (user) {
        currentUserEmail = user.email;
        // Získání jména přihlášeného uživatele z kolekce users
        const userSnap = await getDoc(doc(db, "users", user.email));
        if (userSnap.exists()) {
            currentUserName = userSnap.data().name || userSnap.data().username || user.email;
        } else {
            currentUserName = user.email;
        }
        await loadVehiclesAllocation();
    } else {
        window.location.href = "login.html";
    }
});

async function loadVehiclesAllocation() {
    const container = document.getElementById("vehiclesContainer");
    if (!container) return;
    container.innerHTML = "";

    try {
        const vehiclesSnap = await getDocs(collection(db, "vehicles"));
        const assignmentsSnap = await getDocs(collection(db, "assignments"));
        
        let assignmentsMap = {};
        assignmentsSnap.forEach(d => { assignmentsMap[d.id] = d.data(); });

        if (vehiclesSnap.empty) {
            container.innerHTML = `<p style="opacity:0.5;">Žádná vozidla nebyla v systému nalezena.</p>`;
            return;
        }

        vehiclesSnap.forEach(vDoc => {
            const evc = vDoc.id;
            const assignData = assignmentsMap[evc] || { drivers: [], pendingRequests: [] };
            const assignedDrivers = assignData.drivers || [];
            
            let statusText = "Volný";
            let statusClass = "status-free";
            let disableBtn = false;

            if (assignedDrivers.length === 1) {
                statusText = "Částečně obsazen";
                statusClass = "status-occupied";
            } else if (assignedDrivers.length >= 2) {
                statusText = "Plně obsazen";
                statusClass = "status-full";
                disableBtn = true;
            }

            // Pokud už uživatel má o auto zažádáno, nebo ho má přidělené
            if (assignedDrivers.some(d => d.email === currentUserEmail) || (assignData.pendingRequests && assignData.pendingRequests.some(r => r.email === currentUserEmail))) {
                disableBtn = true;
            }

            let driversHtml = "";
            if (assignedDrivers.length === 0) {
                driversHtml = `<li>❌ Žádný stálý řidič</li>`;
            } else {
                assignedDrivers.forEach(d => {
                    driversHtml += `<li>👤 <strong>${d.name}</strong></li>`;
                });
            }

            const card = document.createElement("div");
            card.className = "vehicle-card";
            card.innerHTML = `
                <h3>🚍 Ev.č. ${evc}</h3>
                <span class="status-badge ${statusClass}">${statusText}</span>
                <ul class="drivers-list">
                    ${driversHtml}
                </ul>
                <button class="btn-request" ${disableBtn ? "disabled" : ""} onclick="requestVehicle('${evc}', ${assignedDrivers.length})">
                    ${assignedDrivers.some(d => d.email === currentUserEmail) ? "Máš přiděleno" : "Zažádat o přidělení"}
                </button>
            `;
            container.appendChild(card);
        });
    } catch (e) { console.error(e); }
}

window.requestVehicle = async function(evc, currentDriversCount) {
    if (currentDriversCount === 1) {
        const potvrditDomluvu = confirm(`⚠️ Na tomto vozidle už jezdí jeden kolega.\n\nOpravdu se s ním předem domlouváš na střídání na voze a chceš odeslat žádost vedení firmy?`);
        if (!potvrditDomluvu) return;
    }

    try {
        const assignRef = doc(db, "assignments", evc);
        const assignSnap = await getDoc(assignRef);
        
        let currentRequests = [];
        let currentDrivers = [];
        if (assignSnap.exists()) {
            currentRequests = assignSnap.data().pendingRequests || [];
            currentDrivers = assignSnap.data().drivers || [];
        }

        currentRequests.push({
            email: currentUserEmail,
            name: currentUserName,
            requestedAt: new Date().toISOString()
        });

        await setDoc(assignRef, {
            drivers: currentDrivers,
            pendingRequests: currentRequests
        }, { merge: true });

        alert(`Žádost o vůz Ev.č. ${evc} byla úspěšně odeslána ke schválení vedení!`);
        await loadVehiclesAllocation();
    } catch (e) { alert("Chyba při odesílání žádosti: " + e.message); }
};
